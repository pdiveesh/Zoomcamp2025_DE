 
//To run postgres locally
 
 docker run -it \
>     -e POSTGRES_USER="root" \
>     -e POSTGRES_PASSWORD="root" \
>     -e POSTGRES_DB="ny_taxi" \
>     -v $(pwd)/1_intro/ny_taxi_postgres_new_data:/var/lib/postgresql/data \
>     -p 5432:5432 \
>     postgres:13



You can now start the database server using:

 to open db:   
pgcli -h localhost -p 5432 -u root -d ny_taxi

//To use pgadmin

docker pull dpage/pgadmin4:<tag name> i.e, version(latest,8.4 etc  )https://www.pgadmin.org/docs/pgadmin4/latest/container_deployment.html
//Running pgadmin


docker run -it \
  -e PGADMIN_DEFAULT_EMAIL="admin@admin.com" \
  -e PGADMIN_DEFAULT_PASSWORD="root" \
  -p 8080:80 \
  dpage/pgadmin4




##NETWORK
docker network create pg-network




  docker run -it \
    -e POSTGRES_USER="root" \
    -e POSTGRES_PASSWORD="root" \
    -e POSTGRES_DB="ny_taxi" \
    -v $(pwd)/1_intro/ny_taxi_postgres_new_data:/var/lib/postgresql/data \
    -p 5432:5432 \
    --network=pg-network \
    --name pg-database \
    postgres:13
